# JS控制对象访问的两种方式

# getter和setter

# 通过对象字面量定义，或在ES6的class中定义。 以下例子（ES6一样）

let person={
  a: 3,
 get b(){     //在b前面添加关键字get定义get方法
  console.log('a被拿了')
  return this.a
 },
 set b(val){  //通过在属性名前添加关键字set定义setter方法。 
  console.log('a被设置')
  this.a=val
 }
}
console.log(person.b,person.b=44) //输出3,44，访问或者设置属性时隐式调用getter或者setter

针对指定的属性不一定需要同时定义getter和setter。例如，通常我们仅提供getter。如果 在这种情况下试图写入属性值，具体的行为取决于代码是在严格模式还是非严格模式。如果在非严格模式下，对仅有getter的属性赋值不起作用，JavaScript引擎会默默地忽略我们的请求。另一方面，如果在严格模式下，JavaScript引擎将会抛出异常，表明我们试图将给一个仅有getter没有setter的属性赋值。

  • 缺点：JavaScript没有私有对象属性。我们可以通过闭包模拟私有属性，通过定义变量和指定对象包含这些变量。但由于对象字面量与类、getter和setter 方法不是在同一个作用域中定义的，因此那些希望作为私有对象属性的变量是无法实现的。

# 通过使用内置的Object.defineProperty方法。

function Person(){
let _skill = 0
Object.defineProperty(this, 'skill', {
	get: ()=>{
		console.log('get触发了')
		return _skill
	},
	set: (val)=>{
		console.log('set触发了')
		_skill = val
	}
})
}
let person = new Person()
console.log(person._skill)  //undefined
console.log(person.skill)   //get触发了,0
console.log(person.skill=10) //set触发了,10
  • 优点：可以使用来进行私有变量，类型校验，计算属性，处理日志等，但是有一些情况我们需要控制对象的全部交互，而getter和setter只能控制对象属性的读取 或者设置。这时候就需要代理来解决。

# 代理Proxy

  • 与以上方式的区别：概念：代理（proxy）是我们通过代理控制对另一个对象的访问。通过代理可以定义当对象发生交互时可执行的自定义行为——如读取或设置属性值，或调用方法。可以将代理理解为通用化的setter与getter，区别是每个setter与getter仅能控制单个对象属性，而代理可用于对象交互的通用处理，包括调用对象的方法。
var emperor = {name:'komei'}  //目标对象
var representative = new Proxy(emperor,{
    get: (target,key) =>{
    	console.log('key的值是'+key)
    	return key in target ? target[key]: '不存在'
    },
    set: (target,key,value) => {
    	console.log('设置'+key+'的值')
    	target[key] = value
    }
})
console.log('直接调用',emperor.name)    //komei
console.log('代理对象',representative.name)  //key的值是name,代理对象 komei
console.log('对象不存在的值',emperor.nam)    //对象不存在的值 undefined
console.log('代理对象不存在的值',representative.nam)  //key的值是nam 代理对象不存在的值 不存在
representative.name='靓仔'
representative.nam='靓仔二号'
console.log('新的对象值',emperor)  //新的对象值 {name: "靓仔", nam: "靓仔二号"}
  • 除了使用get和set，还有许多其他的内置方法用于定义各种对象的行为，如下：

1.调用函数时激活apply，使用new操作符时激活construct。（即将对象作为函数调用时）

2.读取/写入属性时激活get与set。

3.执行for-in语句时激活enumerate。

4.获取和设置属性值时激活getPrototypeOf与setPrototypeOf
...

  • 代理的一些应用场景

1.使用代理记录日志

2.使用代理检测某段代码性能

3.使用代理自动填充属性

4.使用的代理实现负数组索引 ...

以上这些并不一定要使用代理才能实现，但是使用代理可以将业务代码和这些另类代码解耦开来，易于维护。

  • 性能问题

代理的性能比普通的对象控制要差20-50倍

  • 使用代理和普通方法的一个对比例子
假设需要抽象 计算机的文件夹结构模型，一个文件夹对象既可以有属性，也可以是文件夹。现在假设你需要长路径的文件模型,如

rootFolder.ninjasDir.firstNinjaDir.ninjaFile = "yoshi.txt";
普通方法：
const rootFolder = new Folder(); 
rootFolder.ninjasDir = new Folder(); 
rootFolder.ninjasDir.firstNinjaDir = new Folder(); 
rootFolder.ninjasDir.firstNinjaDir.ninjaFile = "yoshi.txt";

代理方式：
function Folder() { 　 
    return new Proxy({}, { 　　 
        get: (target, property) => { 　　　　
        　　　　 if(!(property in target)) {
        　　　　　　target[property] = new Folder(); 　　　　 
              }
    　　　　return target[property]; 　　　
        } 　
    }); 
}
const rootFolder = new Folder();
try { 　
    rootFolder.ninjasDir.firstNinjaDir.ninjaFile = "yoshi.txt";　 //一直触发get方法，一直new Folder对象
    console.log('设置成功')    //成功输出
} 
catch(e){ 　 
    console.log(e);    //没有输出
}
